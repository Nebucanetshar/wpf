var viewport = new HelixViewport3D
{
    Background = Brushes.LightGray
};

// Configurer la caméra
viewport.Camera = new PerspectiveCamera
{
    Position = new Point3D(0, 5, 20),
    LookDirection = new Vector3D(0, -5, -20),
    UpDirection = new Vector3D(0, 1, 0)
};

// Activer la manipulation de la caméra
viewport.CameraController.InertiaFactor = 0.93; // Ajuste la fluidité
viewport.CameraController.RotationSensitivity = 2.0;
viewport.CameraController.ZoomSensitivity = 1.2;
viewport.CameraController.PanSensitivity = 0.5;

// Ajouter un modèle 3D
var sphere = new SphereVisual3D
{
    Center = new Point3D(0, 0, 0),
    Radius = 2,
    Fill = Brushes.Blue
};

viewport.Children.Add(sphere);
